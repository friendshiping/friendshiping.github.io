##2. Spring配置文件






        <!--调用dataSource加载的xml文件创建factory-->
<bean id="factory" class="org.mybatis.spring.SqlSessionFactoryBean">
    <property name="dataSource" ref="dataSource"></property>
</bean>



            <!--factory中对应属性值 通过对扫描包给其赋值-->
<bean  class="org.mybatis.spring.mapper.MapperScannerConfigurer">
    <property name="basePackage" value="com.can.mapper"></property>
    <property name="sqlSessionFactory" ref="factory"></property>
</bean>



        <!--将包扫描的结果生成的类赋值给给定类的属性-->
<bean id="sel" class="com.can.sev.AirportImpl">
    <property name="airportMapper" ref="airportMapper"></property>
</bean>

#注意#

  1. applicationContext.xml建立在src下（classpath默认会从这里开始找）
  2. ApplicationContext ac = new ClassPathXmlApplicationContext(“ApplicationContext.xml”);
    （从classpath路径下找到applicationContext.xml加载）
  3. AirportImpl sel = ac.getBean(“sel”, AirportImpl.class);
    （拿到Spring管理的类生成的对象）
